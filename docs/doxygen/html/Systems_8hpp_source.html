32 inline void handleClientInputs(Registry &r) {
33 Network::INetworkServer &network = Server::getInstance().getNetwork();
34
35 SparseArray<ClientInputs> &inputs = r.get_components<ClientInputs>();
36 SparseArray<Position> &positions = r.get_components<Position>();
37 SparseArray<Velocity> &velocities = r.get_components<Velocity>();
38 SparseArray<Delay> &clocks = r.get_components<Delay>();
39 const SparseArray<Bonus> &bonuses = r.get_components<Bonus>();
40 const SparseArray<Life> &lifes = r.get_components<Life>();
41 const SparseArray<NetworkId> &network_ids = r.get_components<NetworkId>();
42 std::queue<std::function<void()>> callbacks;
43
44 for (auto &&[entity, input, velocity, position, clock, life, bonus] : IndexedZipper(inputs, velocities, positions, clocks, lifes, bonuses)) {
45 if (!life.is_alive())
46 continue;
47 velocity.x = 0;
48 velocity.y = 0;
49 ++clock.last;
50 if (!input.input_keys.empty()) {
51 if (std::ranges::find(input.input_keys, Protocol::MOVE_UP) != input.input_keys.end())
52 velocity.y -= PLAYER_SPEED;
53 else if (std::ranges::find(input.input_keys, Protocol::MOVE_DOWN) != input.input_keys.end())
54 velocity.y += PLAYER_SPEED;
55 if (std::ranges::find(input.input_keys, Protocol::MOVE_LEFT) != input.input_keys.end())
56 velocity.x -= PLAYER_SPEED;
57 else if (std::ranges::find(input.input_keys, Protocol::MOVE_RIGHT) != input.input_keys.end())
58 velocity.x += PLAYER_SPEED;
59 if (std::ranges::find(input.input_keys, Protocol::SHOOT) != input.input_keys.end() && clock.delay <= clock.last) {
60 clock.last = 0;
61 callbacks.emplace([&] {
62 PlayerBullet::create(r, Position(position.x + PLAYER_SIZE_X, position.y + PLAYER_SIZE_Y / 2.0f), bonus);
63 });
64 }
65 }
66 for (auto &&[network_id] : Zipper(network_ids)) {
67 Protocol::EntityPositionVelocityPacket pos_vel(
68 entity,
69 {position.x, position.y},
70 {velocity.x, velocity.y});
71 Network::Packet packet(
72 pos_vel,
73 Protocol::POSITION_VELOCITY
74 );
75 network.send(network_id.id, packet.serialize());
76 }
77
78 std::optional<Position> &bon_pos = positions[bonus.id];
79 std::optional<Velocity> &bon_vel = velocities[bonus.id];
80 if (bonus.type == Bonus::None || !bon_pos || !bon_vel)
81 continue;
82 bon_pos->x = position.x + PLAYER_SIZE_X;
83 bon_pos->y = position.y;
84 bon_vel->x = velocity.x;
85 bon_vel->y = velocity.y;
86
87 for (auto &&[network_id] : Zipper(network_ids)) {
88 Protocol::EntityPositionVelocityPacket pos_vel(
89 bonus.id,
90 {bon_pos->x, bon_pos->y},
91 {bon_vel->x, bon_vel->y});
92 Network::Packet packet(
93 pos_vel,
94 Protocol::POSITION_VELOCITY
95 );
96 network.send(network_id.id, packet.serialize());
97 }
98 }
99 while (!callbacks.empty()) {
100 callbacks.front()();
101 callbacks.pop();
102 }
103 }

105 inline void levelEndlessHandler(Registry &r, std::size_t &score) {
106 if (score == 0) {
107 const SparseArray<NetworkId> &network_ids = r.get_components<NetworkId>();
108 SparseArray<Position> &positions = r.get_components<Position>();
109 const SparseArray<Velocity> &velocities = r.get_components<Velocity>();
110 float i = 100;

112
113 for (auto &&[entity, _, position, velocity] : IndexedZipper(network_ids, positions, velocities)) {

117 Protocol::EntityPositionVelocityPacket pos_vel(entity, {position.x, position.y}, {velocity.x, velocity.y});

120 network.send(network_id.id, packet.serialize());
121 }

125
126 if (score % 60 != 0)
127 return;
128 std::vector<std::pair<std::function<entity_t(Registry &)>, double>> spawn_rates{
129 {EnemyFly::create, 2},
130 {EnemyTurret::create, 1},
131 {EnemyTank::create, 0.3}
132 };
133 double difficulty = static_cast<double>(score) / 500.0 * (0.8 + static_cast<double>(std::rand()) / RAND_MAX * (1.2 - 0.8)) + 0.5;
134 if (difficulty > 1)
135 difficulty = 1 + (difficulty - 1) * 0.5;
136 if (difficulty > 3)
137 difficulty = 3 + (difficulty - 3) * 0.5;
138 if (difficulty > 4)
139 difficulty = 4;
140 for (auto &&[spawn, rate] : spawn_rates) {
141 for ([[maybe_unused]] std::size_t _ : std::ranges::iota_view{static_cast<std::size_t>(0), static_cast<std::size_t>(difficulty * rate)})
142 spawn(r);
143 }

226 void loadFile_() {
227 std::ifstream ifs("assets/levels.txt");
228 std::string line;
229
230 if (!ifs.is_open())
231 throw std::runtime_error("Failed to open levels file: levels.txt");
232
233 std::getline(ifs, line);
234 while (ifs) {
235 if (line.empty() || line[0] == '#') {
236 std::getline(ifs, line);
237 continue;
238 }
239 std::size_t width;
240 std::size_t height;
241 std::stringstream ss(line);
242 ss >> width >> std::skipws >> height;
243 if (!width || !height)
244 throw std::runtime_error("Levels file: Missing width or height");
245 levels_.emplace_back(width, std::vector(height, VOID_ELEM_TYPE));
246 for (std::size_t y = 0; y < height; ++y) {
247 std::getline(ifs, line);
248 if (line.size() > width)
249 throw std::runtime_error("Levels file: Size width missmatch");
250 std::size_t i = 0;
251 for (char c : line)
252 levels_.back()[i++][y] = map_char_elem_type_.contains(c) ? map_char_elem_type_.at(c) : VOID_ELEM_TYPE;
253 }
254 std::getline(ifs, line);
255 }
256 ifs.close();
257 }
